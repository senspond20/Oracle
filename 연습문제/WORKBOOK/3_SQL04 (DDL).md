CREATE OR REPLACE VIEW VW_지도면담
    AS SELECT STUDENT_NAME, DEPARTMENT_NAME, PROFESSOR_NAME
    FROM TB_STUDENT
        JOIN TB_DEPARTMENT USING(DEPARTMENT_NO)
        JOIN TB_PROFESSOR ON (COACH_PROFESSOR_NO = PROFESSOR_NO);
        
SELECT * FROM VW_지도면담;

SELECT * FROM VW_학과별학생수;
